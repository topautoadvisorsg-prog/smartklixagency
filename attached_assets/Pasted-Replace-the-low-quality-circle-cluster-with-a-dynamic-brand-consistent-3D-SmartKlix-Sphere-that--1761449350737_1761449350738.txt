⚙️ COMPONENT CODE (React + Three.js)
// SmartKlixSphere.jsx
import * as THREE from "three";
import { Canvas, useFrame } from "@react-three/fiber";
import { useMemo, useRef } from "react";
import "./SphereStyles.css";

function Nodes() {
  const group = useRef();
  const nodes = useMemo(() => {
    const arr = [];
    for (let i = 0; i < 16; i++) {
      const phi = Math.acos(-1 + (2 * i) / 16);
      const theta = Math.sqrt(16 * Math.PI) * phi;
      arr.push({
        position: [
          Math.cos(theta) * Math.sin(phi),
          Math.sin(theta) * Math.sin(phi),
          Math.cos(phi),
        ],
      });
    }
    return arr;
  }, []);

  useFrame(({ clock }) => {
    const t = clock.getElapsedTime();
    group.current.rotation.y = t / 8;
    group.current.rotation.x = t / 10;
  });

  return (
    <group ref={group}>
      {nodes.map((n, i) => (
        <mesh key={i} position={n.position}>
          <sphereGeometry args={[0.05, 32, 32]} />
          <meshStandardMaterial emissive={"#F4C430"} emissiveIntensity={0.6} color={"#0D1B2A"} />
        </mesh>
      ))}
    </group>
  );
}

export default function SmartKlixSphere() {
  return (
    <div className="sphere-wrapper">
      <Canvas camera={{ position: [0, 0, 3] }}>
        <ambientLight intensity={0.4} />
        <pointLight position={[2, 2, 3]} intensity={1.5} color={"#F4C430"} />
        <Nodes />
      </Canvas>
    </div>
  );
}
